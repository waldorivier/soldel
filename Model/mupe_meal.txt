

CREATE TABLE mupe.meal_content (
   food_id BIGINT(18) NOT NULL,
   meal_id BIGINT(18) NOT NULL,
   modified TINYINT(1) NOT NULL,
   PRIMARY KEY (food_id, meal_id)
);


CREATE TABLE mupe.symptom (
   symptom_id BIGINT(18) NOT NULL,
   symptom_cat BIGINT(18) NOT NULL,
   value LONGTEXT NOT NULL,
   symptom_date DATETIME NOT NULL,
   PRIMARY KEY (symptom_id)
);

CREATE TABLE mupe.meal_symptom (
   meal_id BIGINT(18) NOT NULL,
   symptom_id BIGINT(18) NOT NULL,
   PRIMARY KEY (meal_id, symptom_id)
);

CREATE TABLE mupe.caterory (
   category_id BIGINT(18) NOT NULL,
   name LONGTEXT NOT NULL,
   PRIMARY KEY (category_id)
);


CREATE TABLE mupe.food (
   food_id BIGINT(18) NOT NULL,
   category_fkid BIGINT(18) NOT NULL,
   name LONGTEXT NOT NULL,
   category_id BIGINT(18) NOT NULL,
   PRIMARY KEY (food_id),
   CONSTRAINT FK_food_caterory_0 FOREIGN KEY (category_id) REFERENCES mupe.caterory (category_id)
);


CREATE TABLE mupe.meal (
   meal_id BIGINT(18) NOT NULL,
   meal_date DATETIME NOT NULL,
   meal_code BIGINT(18) NOT NULL,
   meal_code_str LONGTEXT NOT NULL,
  
   PRIMARY KEY (meal_id)
);
